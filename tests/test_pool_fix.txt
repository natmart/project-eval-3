class TestWorkerPoolInitialization:
    """Test WorkerPool initialization and basic setup."""

    def test_pool_initialization_defaults(self) -> None:
        """Test pool initialization with default values."""
        queue = TaskQueue()
        pool = WorkerPool(queue)

        assert pool.queue == queue
        assert pool.worker_count == 4
        assert pool.poll_interval == 0.1
        assert pool.retry_policy is not None
        assert pool.retry_policy.max_attempts == 3
        assert pool.active_workers_count == 0
        assert pool.completed_tasks_count == 0
        assert pool.failed_tasks_count == 0

    def test_pool_initialization_custom_values(self) -> None:
        """Test pool initialization with custom values."""
        queue = TaskQueue()
        retry_policy = RetryPolicy(max_attempts=5, base_delay=2.0)
        pool = WorkerPool(
            queue=queue,
            worker_count=2,
            poll_interval=0.5,
            retry_policy=retry_policy,
        )

        assert pool.worker_count == 2
        assert pool.poll_interval == 0.5
        assert pool.retry_policy.max_attempts == 5
        assert pool.retry_policy.base_delay == 2.0

    def test_pool_creates_expected_workers(self) -> None:
        """Test that pool creates the expected number of workers."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=6)

        metrics = pool.get_worker_metrics()
        assert len(metrics) == 6
        # Workers should not be running initially
        assert pool.active_workers_count == 0

    def test_pool_initial_zero_completed_failed_tasks(self) -> None:
        """Test that pool starts with zero completed and failed task counts."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=4)

        assert pool.completed_tasks_count == 0
        assert pool.failed_tasks_count == 0

        pool_summary = pool.get_pool_summary()
        assert pool_summary["completed_tasks"] == 0
        assert pool_summary["failed_tasks"] == 0

        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=2)

        assert len(initial_metrics) == 2

        scaled_metrics = pool.get_worker_metrics()

        """Test that scaling up works when pool is running."""
        queue = TaskQueue()

        pool.start_all()

        pool.scale_up(2)

        # New workers are not automatically started
        assert pool.active_workers_count == initial_active
        assert len(pool.get_worker_metrics()) == 4

        pool.stop_all(timeout=1.0)

    def test_scale_down_removes_workers(self) -> None:
        """Test that scale_down removes specified number of workers."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=5)

        initial_metrics = pool.get_worker_metrics()
        assert len(initial_metrics) == 5

        pool.scale_down(2)

        scaled_metrics = pool.get_worker_metrics()
        assert len(scaled_metrics) == 3

    def test_scale_down_stops_removed_workers(self) -> None:
        """Test that scaled-down workers are stopped."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=4)

        pool.start_all()
        time.sleep(0.2)
        assert pool.active_workers_count == 4

        pool.scale_down(2)
        time.sleep(0.1)

        # Removed workers should be stopped
        assert pool.active_workers_count == 2

        pool.stop_all(timeout=1.0)

    def test_scale_down_more_than_available(self) -> None:
        """Test scaling down more workers than available safely handles case."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=3)

        pool.start_all()
        time.sleep(0.2)
        assert pool.active_workers_count == 3

        # Try to remove 5 workers when only 3 exist
        pool.scale_down(5)
        time.sleep(0.1)

        # Should only remove available workers
        assert len(pool.get_worker_metrics()) == 0
        assert pool.active_workers_count == 0

        queue = TaskQueue()

    def test_concurrent_operations_safety(self) -> None:
        """Test that concurrent scaling operations are safe."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=2)

        def scale_up_worker():
            for _ in range(5):
                pool.scale_up(1)
                time.sleep(0.01)

        def scale_down_worker():
            for _ in range(3):
                pool.scale_down(1)
                time.sleep(0.01)

        threads = [
            threading.Thread(target=scale_up_worker),
            threading.Thread(target=scale_down_worker),
        ]

        for thread in threads:
            thread.start()

        for thread in threads:
            thread.join()

        # Pool should be in a valid state
        metrics = pool.get_worker_metrics()
        assert len(metrics) >= 0
        assert len(metrics) <= 9  # Initial 2 + 5 up - 3 down max

        pool.stop_all(timeout=1.0)


class TestWorkerPoolThreadPoolSafety:
    """Test WorkerPool thread-safety."""

    def test_concurrent_task_submission(self) -> None:
        """Test that concurrent task submissions work correctly."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=4)

        task_results = []

        def handler(task: Task) -> Any:
            time.sleep(0.01)
            task_results.append(task.payload)
            return True

        pool.register_handler("concurrent_task", handler)
        pool.start_all()

        # Submit tasks from multiple threads
        def submit_tasks(thread_id: int):
            for i in range(10):
                task = Task(name="concurrent_task", payload=f"{thread_id}_{i}")
                queue.enqueue(task)

        threads = [
            threading.Thread(target=submit_tasks, args=(i,)) for i in range(3)
        ]

        for thread in threads:
            thread.start()

        for thread in threads:
            thread.join()

        time.sleep(0.5)

        # All tasks should be completed
        assert len(task_results) == 30
        assert pool.completed_tasks_count == 30

        pool.stop_all(timeout=1.0)

    def test_concurrent_scaling_while_running(self) -> None:
        """Test that scaling works while pool is actively running."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=2)

        def handler(task: Task) -> Any:
            time.sleep(0.05)
            return True

        pool.register_handler("test_task", handler)
        pool.start_all()

        # Add tasks continuously
        def add_tasks():
            for i in range(20):
                task = Task(name="test_task", payload=i)
                queue.enqueue(task)
                time.sleep(0.02)

        # Scale up while running
        def scale_up():
            for _ in range(3):
                pool.scale_up(1)
                time.sleep(0.1)

        add_thread = threading.Thread(target=add_tasks)
        scale_thread = threading.Thread(target=scale_up)

        add_thread.start()
        time.sleep(0.1)
        scale_thread.start()

        add_thread.join()
        scale_thread.join()

        time.sleep(1.0)

        # Verify pool is still functional
        assert pool.completed_tasks_count == 20

        pool.stop_all(timeout=1.0)

    def test_metrics_consistency_under_load(self) -> None:
        """Test that metrics remain consistent under concurrent load."""
        queue = TaskQueue()
        pool = WorkerPool(queue, worker_count=3)

        def success_handler(task: Task) -> Any:
            return True

        def fail_handler(task: Task) -> Any:
            raise Exception("Failed")

        pool.register_handler("success", success_handler)
        pool.register_handler("fail", fail_handler)
        pool.start_all()

        # Add mixed tasks
        for i in range(50):
            if i % 3 == 0:
                queue.enqueue(Task(name="fail", payload=i))
            else:
                queue.enqueue(Task(name="success", payload=i))

        time.sleep(8.0)

        summary = pool.get_pool_summary()
